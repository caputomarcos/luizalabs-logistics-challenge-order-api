/**
 * Representa uma ordem de usuário.
 * @class
 */
class Order {
  /**
   * Construtor para criar uma ordem.
   * @param {number} userId - ID do usuário associado à ordem.
   * @param {string} userName - Nome do usuário associado à ordem.
   */
  constructor(userId, userName) {
    this.user_id = parseInt(userId, 10);
    this.name = userName.trim();
    this.orders = [];
  }

  /**
   * Adiciona uma ordem à lista de ordens do usuário.
   * @param {number} orderId - O ID da ordem.
   * @param {string} date - A data da ordem no formato AAAA-MM-DD.
   * @returns {Object} A ordem criada.
   */
  addOrder(orderId, date) {
    const formattedDate = `${date.substring(0, 4)}-${date.substring(4, 6)}-${date.substring(6, 8)}`;
    const order = {
      order_id: parseInt(orderId, 10),
      date: formattedDate,
      products: [],
      total: 0.0,
    };
    this.orders.push(order);
    return order;
  }

  /**
   * Encontra uma ordem pelo ID.
   * @param {number} orderId - ID da ordem a ser encontrada.
   * @returns {Object|undefined} A ordem, se encontrada; caso contrário, undefined.
   */
  findOrder(orderId) {
    return this.orders.find(
      (order) => order.order_id === parseInt(orderId, 10),
    );
  }
}

module.exports = Order;
